document.forms[0].pass.addEventListener("change", function() {
	let pass = document.forms[0].pass.value;
	let err = document.getElementById("passwordErr");
	let strength = [/[a-z]/.test(pass), /[A-Z]/.test(pass), 
	/\d/.test(pass), pass.length >= 8,
	/[-!@#$%^&*\(\)_=+`~.>,<\/?'";:\\|]/.test(pass)]
	.reduce((sum, num) => sum + num, 0);
	switch (strength) {
		case 0:
		case 1:
			err.innerHTML = "*very weak password";
			err.style.color = "red";
			break;
		case 2:
			err.innerHTML = "*weak password";
			err.style.color = "pink";
			break;
		case 3:
			err.innerHTML = "*good password";
			err.style.color = "orange";
			break;
		case 4:
			err.innerHTML = "*very good password";
			err.style.color = "lightgreen";
			break;
		case 5:
			err.innerHTML = "*excellent password"
			err.style.color = "green";
			break;
	}
})
document.forms[0].email.addEventListener("input", function() {
	document.getElementById("emailErr").innerHTML = "";
});
